% Fuente de figura debajo del pie
% Uso: \fuentefigura{Elaboración propia con base en \citep{garcia2019}.}
\newcommand{\fuentefigura}[1]{%
  \par\vspace{2pt}{\small\textit{Fuente: #1}}%
}

% Nota al margen para revisión (comentar antes de entrega final)
% Uso: \pendiente{Completar con datos de campo}
\newcommand{\pendiente}[1]{%
  \marginpar{\footnotesize\color{alertaRojo}\textbf{[Pendiente:} #1\textbf{]}}%
}

% Resaltado de término definido por primera vez
% Uso: \termino{fragmentación urbana}
\newcommand{\termino}[1]{\textit{#1}}
