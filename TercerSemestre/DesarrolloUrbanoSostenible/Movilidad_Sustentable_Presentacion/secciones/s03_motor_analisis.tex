\section{El Motor de Análisis: Eficiencia e Impedancia}

% Diapositiva 1: Enfoque Conceptual (Sin mencionar VFT)
\begin{frame}{Motor de Análisis: Eficiencia Vial y Accesibilidad}
    \begin{block}{Cambio de Paradigma}
        Transición del enfoque de \textbf{flujo vehicular} (capacidad de carga) al enfoque de 
        \textbf{accesibilidad} (interacción de personas) \cite{ITDP_Espacio}.
    \end{block}
    
    \begin{itemize}
        \item \textbf{Gestión de Movilidad:} Priorización de la eficiencia del sistema sobre la 
              velocidad del objeto \cite{ITDP_Espacio}.
        \item \textbf{Impedancia Sistémica:} Factores físicos y operativos que restringen el 
              derecho humano a la movilidad.
        \item \textbf{Objetivo:} Evaluar la capacidad real de la infraestructura ante la demanda actual.
    \end{itemize}
\end{frame}

% Diapositiva 3: Impacto Económico y Externalidades
\begin{frame}{Fundamentación: Costos de la Ineficiencia Vial}
    \begin{alertblock}{Impacto Macroeconómico}
        Las externalidades (congestión, siniestros y contaminación) cuestan el \textbf{4\% del PIB} nacional
         \cite{IMCO2019Doc, ITDP_Espacio}.
    \end{alertblock}

    \begin{itemize}
        \item \textbf{Productividad:} Pérdida masiva de horas-persona en traslados periféricos.
        \item \textbf{Salud Pública:} Costos por enfermedades respiratorias derivadas de emisiones 
              vehiculares \cite{ITDP_Espacio}.
        \item \textbf{Segregación:} El costo del transporte consume hasta el 19\% del ingreso en el 4to contorno.
    \end{itemize}
\end{frame}
