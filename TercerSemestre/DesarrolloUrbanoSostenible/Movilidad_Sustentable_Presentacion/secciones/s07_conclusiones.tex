\section{Conclusiones}

% ── Frame: Síntesis ───────────────────────────────────────────────────────────
\begin{frame}{Síntesis de Hallazgos}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Lo que funciona}
        \begin{itemize}
          \item El modelo TOD es eficaz donde existe transporte masivo consolidado.
          \item La rehabilitación de barrio genera cohesión social medible.
          \item Los corredores urbanos son compatibles con el marco normativo vigente.
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Los retos pendientes}
        \begin{itemize}
          \item Ausencia de instrumentos de captura de plusvalías.
          \item Fragmentación institucional entre municipios y alcaldías.
          \item Débil participación comunitaria en decisiones de uso de suelo.
        \end{itemize}
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}
