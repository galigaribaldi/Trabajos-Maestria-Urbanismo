\usetheme{Madrid}

% Sobreescribir paleta de Madrid con ColorPrincipal
\setbeamercolor{palette primary}   {bg=ColorPrincipal, fg=white}
\setbeamercolor{palette secondary} {bg=ColorPrincipal!85!black, fg=white}
\setbeamercolor{palette tertiary}  {bg=ColorPrincipal!70!black, fg=white}
\setbeamercolor{palette quaternary}{bg=ColorPrincipal, fg=white}
\setbeamercolor{structure}         {fg=ColorPrincipal}

% Título y frame titles
\setbeamercolor{title}      {fg=white, bg=ColorPrincipal}
\setbeamercolor{frametitle} {fg=white, bg=ColorPrincipal}

% Bloque estándar
\setbeamercolor{block title}{fg=white, bg=ColorPrincipal}
\setbeamercolor{block body} {fg=ColorTexto, bg=ColorFondo}

% Bloque de alerta
\setbeamercolor{block title alerted}{fg=white, bg=ColorAlerta}
\setbeamercolor{block body alerted} {fg=ColorTexto, bg=ColorFondo}

% Bloque de ejemplo
\setbeamercolor{block title example}{fg=white, bg=ColorPrincipal!60!black}

% Acento en texto de énfasis
\setbeamercolor{alerted text}{fg=ColorAcento!80!black}

% Itemize bullets
\setbeamercolor{item}{fg=ColorPrincipal}
\setbeamercolor{subitem}{fg=ColorAcento!80!black}


% =============================================================================
%  3. PERSONALIZACIÓN DE FOOTLINE (pie de diapositiva)
%     Madrid genera: [Autor | Título] [Sección] [N/Total]
%     Aquí se redefine para mostrar: [Autor] | [Materia] | [N/Total]
% =============================================================================
\makeatletter
\setbeamertemplate{footline}{%
  \leavevmode%
  \hbox{%
    \begin{beamercolorbox}[wd=.45\paperwidth, ht=2.5ex, dp=1.5ex, leftskip=0.4cm]{palette primary}%
      \usebeamerfont{author in head/foot}\insertshortauthor
    \end{beamercolorbox}%
    \begin{beamercolorbox}[wd=.40\paperwidth, ht=2.5ex, dp=1.5ex, center]{palette secondary}%
      \usebeamerfont{title in head/foot}\MateriaCorta
    \end{beamercolorbox}%
    \begin{beamercolorbox}[wd=.15\paperwidth, ht=2.5ex, dp=1.5ex, rightskip=0.4cm, right]{palette primary}%
      \usebeamerfont{date in head/foot}%
      \insertframenumber{} / \inserttotalframenumber
    \end{beamercolorbox}%
  }%
  \vskip0pt%
}
\makeatother


% =============================================================================
%  4. HEADLINE — franja de secciones (estilo Madrid simplificado)
% =============================================================================
\setbeamertemplate{headline}{%
  \leavevmode%
  \begin{beamercolorbox}[wd=\paperwidth, ht=1.5ex, dp=0.5ex]{palette tertiary}%
    \hspace{0.4cm}{\tiny \insertsectionhead}%
  \end{beamercolorbox}%
}


% =============================================================================
%  5. FRAMETITLE CON LÍNEA DORADA
% =============================================================================
\setbeamertemplate{frametitle}{%
  \vspace{0.1cm}%
  {\bfseries\insertframetitle}\\[-0.05cm]%
  {\color{ColorAcento}\rule{\textwidth}{1.5pt}}%
}

% =============================================================================
%  7. DIAPOSITIVA DE SECCIÓN
%     Solo aparece a partir de la SEGUNDA sección.
%     La primera sección arranca directo (el TOC ya fue la transición).
% =============================================================================
\newcounter{secciones}
\setcounter{secciones}{0}

\AtBeginSection[]{%
  \addtocounter{secciones}{1}%
  \ifnum\value{secciones}>1
    \begin{frame}[plain, noframenumbering]
      \begin{center}
        \vfill
        {\color{ColorAcento}\rule{0.6\textwidth}{1.5pt}}\\[0.4cm]
        {\Large\bfseries\color{ColorPrincipal}\insertsectionhead}\\[0.4cm]
        {\color{ColorAcento}\rule{0.6\textwidth}{1.5pt}}
        \vfill
      \end{center}
    \end{frame}
  \fi
}
