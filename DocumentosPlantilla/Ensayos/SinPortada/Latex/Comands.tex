% =============================================================================
%  5. BLOQUE DE TÍTULO SIN PORTADA
%     Genera el encabezado del ensayo en la primera página de texto.

% =============================================================================
\newcommand{\cabeceroensayo}{%
  \thispagestyle{primerapagina}%
  \vspace*{0.4cm}%
  \noindent{\Large\bfseries \TituloEnsayo}\\[0.05cm]%
  \noindent{\large\itshape\color{ColorTexto} \SubtituloEnsayo}\\[0.15cm]%
  {\color{ColorPrincipal}\rule{\textwidth}{2pt}}\\[0.25cm]%
  \noindent{\small%
    \textbf{\Autor}\ \ ·\ \ \Materia\ \ ·\ \ \Semestre\ \ ·\ \ Profesor:\ \ProfesorAsignado%
  }%
  \vspace{0.6cm}\par%
}
