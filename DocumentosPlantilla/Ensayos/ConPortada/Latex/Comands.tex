% Fuente debajo de figura
% Uso: \fuentefigura{Elaboración propia con base en \citep{autor2024}.}
\newcommand{\fuentefigura}[1]{%
  \par\vspace{2pt}{\small\textit{Fuente: #1}}%
}

% Nota de revisión al margen (comentar antes de entrega)
% Uso: \pendiente{Completar con datos de campo}
\newcommand{\pendiente}[1]{%
  \marginpar{\footnotesize\color{ColorAlerta}\textbf{[Pendiente:} #1\textbf{]}}%
}

% Término definido por primera vez
% Uso: \termino{fragmentación urbana}
\newcommand{\termino}[1]{\textit{#1}}

% Bloque de cita destacada
% Uso: \citaclave{Texto de la definición clave.}
\newcommand{\citaclave}[1]{%
  \begin{quote}
    \color{ColorPrincipal}\itshape #1
  \end{quote}%
}
